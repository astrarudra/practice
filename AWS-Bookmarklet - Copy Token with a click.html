                javascript:(function(){
                    var AWSStr = JSON.stringify({
                      AWS_ACCESS_KEY_ID: document.getElementById("accessKeyId").value,

                      AWS_SESSION_TOKEN: document.getElementById("sessionToken").value,
                    });
                    AWSStr = (AWSStr.substring(1,AWSStr.length-1)+',').replaceAll('",','",\n');
                    document.getElementsByClassName("pane")[0].innerHTML = "<div><h3>Option Solar: Custom From Bookmarklet <span style='font-size:20px;'>😀</span></h3><br/><textarea class='code-section' id='tocopy' style='width: 642px; height: 62px;'>"+AWSStr+"</textarea><a class='copy-value' id='copyButton'>Copy</a></div>";
                    var el = document.getElementById("copyButton");
                    el.onclick = function() {
                      navigator.clipboard.writeText(document.getElementById("tocopy").value);
                    };
                    el.click();
                  })();
